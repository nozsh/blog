:: config.txt

  for /f "tokens=1,* delims==" %%a in (config.txt) do (
    set %%a=%%b
  )
)

:: 7z
if not defined z7dir (
  echo - Path to 7z dir:
  set /p z7dir="#: "
)

  echo 7z dir - "%z7dir%"
  echo Not be found...
  pause
  exit
)

  set /p archive="#: "

  echo Not be found...
  pause
  exit
)

  echo Not be found...
  pause
  exit

:yesornot

  exit

  echo ERROR!
  echo.
  call :yesornot

  set pwd=%%a
  call :try
)

exit
